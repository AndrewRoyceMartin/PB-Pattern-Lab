  const winner = result.winner;

  if (!winner) {
    return {
      strategy: "Random",
      confidenceTier: "NONE",
      reason: "No valid strategies."
    };
  }

  return {
    strategy: winner.name,
    confidenceTier: winner.confidenceTier,
    reason: `Best validation score: ${winner.name}`
  };
}
