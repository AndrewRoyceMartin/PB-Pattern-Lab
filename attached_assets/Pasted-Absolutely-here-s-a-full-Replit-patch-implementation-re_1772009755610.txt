function shallowComparableConfig(config: ValidationBenchmarkConfig) {
  return {
    windows: [...config.windows].sort((a,b)=>a-b),
    benchmarkMode: config.benchmarkMode,
    enableRegimeSplits: config.enableRegimeSplits,
    seed: config.seed,
    randomBaselineRuns: config.randomBaselineRuns,
    runPermutation: config.runPermutation,
    permutationRuns: config.permutationRuns,
    selectedStrategies: [...config.selectedStrategies].sort(),
    preset: config.preset,
  };
}

function configsMatch(ui: ValidationBenchmarkConfig, used: any) {
  if (!used) return true;
  return JSON.stringify(shallowComparableConfig(ui)) === JSON.stringify({
    windows: [...(used.windows ?? [])].sort((a:number,b:number)=>a-b),
    benchmarkMode: used.benchmarkMode,
    enableRegimeSplits: used.enableRegimeSplits,
    seed: used.seed,
    randomBaselineRuns: used.randomBaselineRuns,
    runPermutation: used.runPermutation,
    permutationRuns: used.permutationRuns,
    selectedStrategies: [...(used.selectedStrategies ?? [])].sort(),
    preset: used.preset ?? "custom",
  });
}

Display warning if mismatch:

{lastBenchmarkResult?.runConfigUsed && !configsMatch(benchmarkConfig, lastBenchmarkResult.runConfigUsed) && (
  <div className="rounded-lg border border-amber-500/40 bg-amber-500/10 p-3 text-amber-300">
    Warning: Current UI settings differ from the benchmark run shown below.
    Results and exports reflect the <strong>Run Configuration Used</strong>, not the current controls.

function getValidationBadge(strategy: any) {
  // Existing stability / avgDelta logic
  // e.g. possible_edge, weak_edge, no_edge, underperforming
}

function getSignificanceBadge(strategyName: string, permutationTests: any[]) {
  const p = permutationTests?.find(t => t.strategyName === strategyName);
  if (!p) return { label: "Not Tested", tone: "neutral" };

  const pval = p.empiricalPValue;
  if (pval <= 0.05) return { label: "Supported", tone: "green" };
  if (pval <= 0.20) return { label: "Suggestive", tone: "amber" };
  return { label: "Unsupported", tone: "slate" };
}

Backend Patch Example (Route Skeleton)

If you want a concrete route pattern, here’s the key structure:

app.post("/api/validation/benchmark", async (req, res) => {
  try {
    // 1) Parse + normalize (prefer zod if already added)
    const windows = Array.isArray(req.body.windows) ? req.body.windows : [20,40,60,100];
    const benchmarkMode = req.body.benchmarkMode === "rolling_walk_forward"
      ? "rolling_walk_forward"
      : "fixed_holdout";

    const enableRegimeSplits = !!req.body.enableRegimeSplits;
    const seed = Number.isFinite(req.body.seed) ? Number(req.body.seed) : 42;
    const randomBaselineRuns = Math.min(Math.max(Number(req.body.randomBaselineRuns) || 200, 10), 5000);
    const runPermutation = !!req.body.runPermutation;
    const permutationRuns = Math.min(Math.max(Number(req.body.permutationRuns) || 200, 10), 5000);

    const selectedStrategies = Array.isArray(req.body.selectedStrategies)
      ? req.body.selectedStrategies
      : [];

    const preset = typeof req.body.preset === "string" ? req.body.preset : "custom";

    const runConfigUsed = {
      windows,
      benchmarkMode,
      enableRegimeSplits,
      seed,
      randomBaselineRuns,
      runPermutation,
      permutationRuns,
      selectedStrategies,
      preset,
      generatedAt: new Date().toISOString(),
    };

    // 2) Call engine with normalized values only
    const benchmarkResult = runMultiWindowBenchmark({
      windows,
      benchmarkMode,
      enableRegimeSplits,
      seed,
      randomBaselineRuns,
      runPermutation,
      permutationRuns,
      selectedStrategies,
    });

    const benchmarkRunId = `bench_${Date.now()}`;

    // 3) Persist optional DB record here (if implemented)
    // saveBenchmarkRun({ benchmarkRunId, runConfigUsed, benchmarkResult })

    // 4) Return exact run metadata used
    return res.json({
      benchmarkRunId,
      ...benchmarkResult,
      runConfigUsed,
    });

  } catch (err) {
    console.error("Benchmark route error:", err);
    return res.status(500).json({ error: "Benchmark failed" });
  }
});
